    <script>
        function openTab(evt, tabName) {
            var i, tabcontent, tablinks;
            tabcontent = document.getElementsByClassName("tab-content");
            for (i = 0; i < tabcontent.length; i++) { tabcontent[i].classList.add("hidden"); }
            tablinks = document.getElementsByClassName("tab-link");
            for (i = 0; i < tablinks.length; i++) {tablinks[i].classList.remove("border-cyan-400", "text-white");
                tablinks[i].classList.add("border-transparent", "text-gray-500");
            }
            document.getElementById(tabName).classList.remove("hidden");
            evt.currentTarget.classList.add("border-cyan-400", "text-white");
            evt.currentTarget.classList.remove("border-transparent", "text-gray-500");
        }
    </script>
